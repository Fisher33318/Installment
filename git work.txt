You can also upload existing files from your computer using the instructions below.


Git global setup
git config --global user.name "Fisher33318"
git config --global user.email "b05611019@ntu.edu.tw"

Create a new repository
git clone https://gitwork.ypcloud.com/clouder-17/c17-yao.git
cd c17-yao
touch README.md
git add README.md
git commit -m "add README"
git push -u origin master

Push an existing folder
cd existing_folder
git init
git remote add origin https://gitwork.ypcloud.com/clouder-17/c17-yao.git
git add .
git commit -m "Initial commit"
git push -u origin master

Push an existing Git repository
cd existing_repo
git remote rename origin old-origin
git remote add origin https://gitwork.ypcloud.com/clouder-17/c17-yao.git
git push -u origin --all
git push -u origin --tags
